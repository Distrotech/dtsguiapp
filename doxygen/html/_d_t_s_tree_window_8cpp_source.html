307 void DTSTreeWindowEvent::TreeEvent(wxDataViewEvent &event) {
308  DTSTreeWindow *tw;
309  wxDataViewItem p_cont, f_item, l_item;
310  int evid, cnt;
311  bool cont,first = true,last = true;
312 
313  tw = (DTSTreeWindow*)parent;
314  evid = event.GetEventType();
315 
316  if (evid == wxEVT_DATAVIEW_SELECTION_CHANGED) {
317  a_item = event.GetItem();
318  TreeCallback(a_item, DTSGUI_TREE_CB_SELECT);
319  if ((vm->GetNodeID(a_item) == -1) && vm->IsContainer(a_item)) {
320  tree->Expand(a_item);
321  }
322  } else if (evid == wxEVT_DATAVIEW_ITEM_EXPANDED) {
323  parent->TreeResize();
324  } else if (evid == wxEVT_DATAVIEW_ITEM_CONTEXT_MENU) {
325  DTSDVMListStore *d_p, *d_i, *d_c;
326  bool c_sort, del;
327 
328  if ((a_item = event.GetItem())) {
329  if (vm->IsContainer(a_item)) {
330  a_cont = a_item;
331  p_cont = vm->GetParent(a_item);
332  cont = true;
333  } else {
334  a_cont = vm->GetParent(a_item);
335  p_cont = a_cont;
336  cont = false;
337  }
338 
339  d_p = (DTSDVMListStore*)p_cont.GetID();
340  d_c = (DTSDVMListStore*)a_cont.GetID();
341  cnt=vm->GetChildCount(p_cont);
342  if (p_cont && d_p->can_sort && (cnt > 1)) {
343  f_item = vm->GetNthChild(p_cont, 0);
344  if (f_item != a_item) {
345  first = false;
346  }
347  l_item = vm->GetNthChild(p_cont, cnt-1);
348  if (l_item != a_item) {
349  last = false;
350  }
351  }
352 
353  c_sort = (d_p && d_p->can_sort) || (d_c && d_c->can_sort);
354 
355  d_i = (DTSDVMListStore*)a_item.GetID();
356  del = d_i->can_delete;
357  tw->ShowRMenu(cont, vm->GetChildCount(a_cont), first, last, c_sort, del);
358  }
359  } else if (evid == wxEVT_DATAVIEW_ITEM_START_EDITING) {
360  a_item = event.GetItem();
361  DTSDVMListStore *data = (DTSDVMListStore*)event.GetItem().GetID();
362  if (!data->can_edit) {
363  event.Veto();
364  if (data->IsContainer()) {
365  if (tree->IsExpanded(a_item)) {
366  tree->Collapse(a_item);
367  } else {
368  tree->Expand(a_item);
369  }
370  }
371  }
372  } else if (evid == wxEVT_DATAVIEW_ITEM_EDITING_DONE) {
373 #ifndef _WIN32
374  DTSDVMListStore *data = (DTSDVMListStore*)event.GetItem().GetID();
375  event.SetValue(data->GetTitle());
376 #endif
377  parent->SetPaneTitle(event.GetValue());
378  TreeCallback(event.GetItem(), DTSGUI_TREE_CB_EDIT);
379  }

382 void DTSTreeWindowEvent::SortParent(const wxDataViewItem parent, int action) {
383  wxDataViewItemArray expanded, selected;
384  DTSDVMListStore *data;
385  unsigned int cont_cnt, cnt;
386 
387  /* Get enabled/selection state of parent*/
388  tree->GetSelections(selected);
389 
390  /*mark expanded containers*/
391  vm->SetExpanded(parent, tree->IsExpanded(parent));
392  cont_cnt = vm->GetContainers(parent, expanded, false);
393  for (cnt=0; cnt < cont_cnt;cnt++) {
394  vm->SetExpanded(expanded[cnt], tree->IsExpanded(expanded[cnt]));
395  }
396 
397  switch(action) {
398  case DTS_TREEWIN_MENU_MOVEDOWN:
399  vm->MoveChildDown(a_item);
400  break;
401  case DTS_TREEWIN_MENU_MOVEUP:
402  vm->MoveChildUp(a_item);
403  break;
404  case DTS_TREEWIN_MENU_SORT:
405  vm->SortChildren(a_cont);
406  break;
407  }
408 
409  /*re select all selected*/
410  for (cnt=0; cnt < selected.size();cnt++) {
411  tree->Select(wxDataViewItem(selected[cnt]));
412  }
413 
414  /*expand root if was expanded*/
415  data = (DTSDVMListStore*)parent.GetID();
416  if (data && data->IsExpanded()) {
417  tree->Expand(parent);
418  data->SetExpanded(false);
419  }
420 
421  /*re expand containers colapsed*/
422  for (cnt=0; cnt < expanded.size();cnt++) {
423  data = (DTSDVMListStore*)expanded[cnt].GetID();
424  if (data && data->IsExpanded()) {
425  tree->Expand(expanded[cnt]);
426  data->SetExpanded(false);
427  }
428  }
