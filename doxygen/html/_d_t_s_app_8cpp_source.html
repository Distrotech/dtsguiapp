168 int dtsgui_curl::curl_progress_function(void *data, double dltotal, double dlnow, double ultotal, double ulnow) {
169  class curl_progress *cp = (class curl_progress*)data;
170  DTSFrame *f;
171  int val, tot, cur;
172 
173  if (!cp || cp->pause || !(f = cp->GetFrame())) {
174  return 0;
175  }
176 
177  tot = dltotal+ultotal;
178  cur = ulnow+dlnow;
179 
180  val = (tot && cur) ? (1000/tot) * cur : 0;
181  val = ceil(val);
182 
183  if (!cp->pd) {
184  cp->pd = f->StartProgress("Web Transfer Progress", 1000, 1);
185  };
186 
187  if (cp->pd && val && !f->UpdateProgress(val, wxEmptyString)) {
188  return 1;
189  }
190 
191  return 0;
192 }
