462 int DTSFrame::pwevent(struct dtsgui *dtsgui, dtsgui_pane p, int type, int event, void *data) {
463  struct basic_auth *auth;
464  DTSDialog *dp = (DTSDialog*)p;
465 
466  if (event == wx_PANEL_EVENT_BUTTON_YES) {
467  if (!(auth = (struct basic_auth*)dp->GetUserData())) {
468  return 0;
469  }
470  if (auth->user) {
471  free((void*)auth->user);
472  auth->user = NULL;
473  }
474  if (auth->passwd) {
475  memset((void*)auth->passwd, 0, strlen(auth->passwd));
476  free((void*)auth->passwd);
477  auth->passwd = NULL;
478  }
479 
480  auth->user = dp->FindValue("uname");
481  auth->passwd = dp->FindValue("pwd");
482  objunref(auth);
483  }
484  return 1;
485 }
